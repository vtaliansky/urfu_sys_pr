# ER-диаграмма

```mermaid
erDiagram
    КОРПУС ||--o{ НОМЕР : содержит
    КОРПУС }o--o{ УСЛУГА : предоставляет
    КЛИЕНТ ||--o{ БРОНЬ : оформляет
    ОРГАНИЗАЦИЯ ||--o{ ДОГОВОР : заключает
    ДОГОВОР ||--o{ БРОНЬ : применяется
    БРОНЬ ||--o{ НОМЕР_В_БРОНИ : включает
    НОМЕР ||--o{ НОМЕР_В_БРОНИ : бронируется
    КЛИЕНТ ||--o{ ПРОЖИВАНИЕ : проживает
    НОМЕР ||--o{ ПРОЖИВАНИЕ : используется
    БРОНЬ ||--o{ ПРОЖИВАНИЕ : становится
    ПРОЖИВАНИЕ ||--o{ НАЧИСЛЕНИЕ : имеет
    УСЛУГА ||--o{ НАЧИСЛЕНИЕ : начисляется
    КЛИЕНТ ||--o{ ОПЛАТА : оплачивает
    КЛИЕНТ ||--o{ ОТЗЫВ_ЖАЛОБА : оставляет
    ПРОЖИВАНИЕ ||--o{ ОТЗЫВ_ЖАЛОБА : относится
    КОРПУС ||--o{ РАСХОД : имеет

    КОРПУС {
        int id
        string название
        int класс_отеля
        int количество_этажей
        int всего_комнат
        int комнат_на_этаже
    }

    НОМЕР {
        int id
        string номер
        int этаж
        int вместимость
        string тип
        decimal базовая_стоимость
        string статус
    }

    УСЛУГА {
        int id
        string название
        string тип
        decimal стоимость
        boolean платная
    }

    КЛИЕНТ {
        int id
        string ФИО
        string телефон
        string email
        string документ
        date дата_регистрации
    }

    ОРГАНИЗАЦИЯ {
        int id
        string название
        string тип
        string контактное_лицо
        string телефон
        string email
    }

    ДОГОВОР {
        int id
        date дата_начала
        date дата_окончания
        decimal скидка
        string условия
        string статус
    }

    БРОНЬ {
        int id
        date дата_заезда
        date дата_выезда
        int количество_людей
        string статус
        decimal итоговая_стоимость
    }

    НОМЕР_В_БРОНИ {
        int id
        int количество_людей_в_номере
    }

    ПРОЖИВАНИЕ {
        int id
        date дата_заселения
        date дата_выезда
        string статус
    }

    НАЧИСЛЕНИЕ {
        int id
        date дата
        string описание
        decimal сумма
        string статус_оплаты
    }

    ОПЛАТА {
        int id
        date дата
        decimal сумма
        string назначение
    }

    ОТЗЫВ_ЖАЛОБА {
        int id
        date дата
        string тип
        string тема
        string текст
        string статус
    }

    РАСХОД {
        int id
        date дата
        string категория
        decimal сумма
        string описание
    }
```
